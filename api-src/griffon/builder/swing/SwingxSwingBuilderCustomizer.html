16 package griffon.builder.swing;
17 
18 import griffon.inject.DependsOn;
19 import groovy.swing.SwingXBuilder;
20 import groovy.util.Factory;
21 import org.codehaus.griffon.runtime.groovy.view.AbstractBuilderCustomizer;
22 
23 import javax.inject.Named;
24 import java.util.LinkedHashMap;
25 import java.util.Map;
26 
27 /**
28  * @author Andres Almiray
29  */
30 @DependsOn("swing")
31 @Named("swingx-swing")
32 public class SwingxSwingBuilderCustomizer extends AbstractBuilderCustomizer {
33     public SwingxSwingBuilderCustomizer() {
34         SwingXBuilder builder = new SwingXBuilder();
35 
36         Map<String, Factory> factories = new LinkedHashMap<>();
37         for (Map.Entry<String, Factory> e : builder.getFactories().entrySet()) {
38             String name = e.getKey();
39             if (name.startsWith("classic")) continue;
40             factories.put("jx" + name, e.getValue());
41         }
42         setFactories(factories);
43         setVariables(builder.getVariables());
44         setMethods(builder.getExplicitMethods());
45         setProps(builder.getExplicitProperties());
46         setAttributeDelegates(builder.getAttributeDelegates());
47         setPreInstantiateDelegates(builder.getPreInstantiateDelegates());
48         setPostInstantiateDelegates(builder.getPostInstantiateDelegates());
49         setPostNodeCompletionDelegates(builder.getPostNodeCompletionDelegates());
50         setDisposalClosures(builder.getDisposalClosures());
51         setMethodMissingDelegate(builder.getMethodMissingDelegate());
52         setPropertyMissingDelegate(builder.getPropertyMissingDelegate());
53     }
54 }
